部署余额查询代理

为什么需要代理？
浏览器安全策略(CORS)阻止直接调用 AI 服务商 API。通过代理服务器中转可解决此问题。
方案一：Cloudflare Workers (推荐，免费)
export default { async fetch(request, env) { const url = new URL(request.url); const corsHeaders = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'POST, OPTIONS', 'Access-Control-Allow-Headers': 'Content-Type, Authorization', }; if (request.method === 'OPTIONS') { return new Response(null, { headers: corsHeaders }); } if (url.pathname === '/api/balance' && request.method === 'POST') { const { provider, apiKey } = await request.json(); try { let balance = null, currency = 'CNY'; if (provider === 'openai') { const res = await fetch('https://api.openai.com/dashboard/billing/credit_grants', { headers: { 'Authorization': `Bearer ${apiKey}` } }); const data = await res.json(); balance = data.total_available || 0; currency = 'USD'; } else if (provider === 'deepseek') { const res = await fetch('https://api.deepseek.com/user/balance', { headers: { 'Authorization': `Bearer ${apiKey}` } }); const data = await res.json(); balance = data.balance_info?.total_balance || 0; } else if (provider === 'moonshot') { const res = await fetch('https://api.moonshot.cn/v1/users/me/balance', { headers: { 'Authorization': `Bearer ${apiKey}` } }); const data = await res.json(); balance = data.data?.balance || 0; } return new Response(JSON.stringify({ success: true, balance, currency, timestamp: new Date().toISOString() }), { headers: { ...corsHeaders, 'Content-Type': 'application/json' }}); } catch (err) { return new Response(JSON.stringify({ success: false, error: err.message }), { status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' }}); } } return new Response('Not Found', { status: 404 }); } }
方案二：Node.js
const express = require('express'); const cors = require('cors'); const app = express(); app.use(cors()); app.use(express.json()); app.post('/api/balance', async (req, res) => { const { provider, apiKey } = req.body; try { let balance = null, currency = 'CNY'; if (provider === 'openai') { const r = await fetch('https://api.openai.com/dashboard/billing/credit_grants', { headers: { 'Authorization': `Bearer ${apiKey}` } }); balance = (await r.json()).total_available || 0; currency = 'USD'; } else if (provider === 'deepseek') { const r = await fetch('https://api.deepseek.com/user/balance', { headers: { 'Authorization': `Bearer ${apiKey}` } }); balance = (await r.json()).balance_info?.total_balance || 0; } else if (provider === 'moonshot') { const r = await fetch('https://api.moonshot.cn/v1/users/me/balance', { headers: { 'Authorization': `Bearer ${apiKey}` } }); balance = (await r.json()).data?.balance || 0; } res.json({ success: true, balance, currency }); } catch (e) { res.status(500).json({ success: false, error: e.message }); } }); app.listen(3000);
部署步骤
  1. 注册 Cloudflare 账号，进入 Workers & Pages
  2. 创建 Service Worker，粘贴上方代码
  3. 部署后获得 https://xxx.workers.dev 地址
  4. 在本页"同步设置"中填入 https://xxx.workers.dev/api/balance
  5. 为 DeepSeek / Kimi 模型填入 API Key
  6. 点击"测试连接"验证，然后刷新余额
