部署余额查询代理

为什么需要代理？

各 AI 服务商的余额查询 API 存在 CORS 限制，且直接在前端暴露 API Key 有安全风险。通过自建代理服务器可以安全地查询余额。

方案一：Cloudflare Workers (推荐)

免费、全球部署、无需服务器

export default {
  async fetch(request, env) {
    const url = new URL(request.url);
    const corsHeaders = {
      'Access-Control-Allow-Origin': '*',
      'Access-Control-Allow-Methods': 'POST, OPTIONS',
      'Access-Control-Allow-Headers': 'Content-Type',
    };
    if (request.method === 'OPTIONS') {
      return new Response(null, { headers: corsHeaders });
    }
    if (url.pathname === '/api/balance' && request.method === 'POST') {
      const { provider, apiKey, customUrl } = await request.json();
      try {
        let balance = null;
        let currency = 'CNY';
        switch(provider) {
          case 'openai':
            const openaiRes = await fetch('https://api.openai.com/dashboard/billing/credit_grants', {
              headers: { 'Authorization': `Bearer ${apiKey}` }
            });
            const openaiData = await openaiRes.json();
            balance = openaiData.total_available || 0;
            currency = 'USD';
            break;
          case 'deepseek':
            const dsRes = await fetch('https://api.deepseek.com/user/balance', {
              headers: { 'Authorization': `Bearer ${apiKey}` }
            });
            const dsData = await dsRes.json();
            balance = dsData.balance_info?.total_balance || 0;
            break;
          case 'moonshot':
            const mkRes = await fetch('https://api.moonshot.cn/v1/users/me/balance', {
              headers: { 'Authorization': `Bearer ${apiKey}` }
            });
            const mkData = await mkRes.json();
            balance = mkData.data?.balance || 0;
            break;
          default:
            if (customUrl) {
              const customRes = await fetch(customUrl, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ provider, apiKey })
              });
              const customData = await customRes.json();
              balance = customData.balance;
              currency = customData.currency || 'CNY';
            }
        }
        return new Response(JSON.stringify({ 
          success: true, balance, currency,
          timestamp: new Date().toISOString()
        }), { headers: { ...corsHeaders, 'Content-Type': 'application/json' }});
      } catch (err) {
        return new Response(JSON.stringify({ 
          success: false, error: err.message 
        }), { status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' }});
      }
    }
    return new Response('Not Found', { status: 404 });
  }
}

方案二：Node.js 服务器

const express = require('express');
const cors = require('cors');
const fetch = require('node-fetch');
const app = express();
app.use(cors());
app.use(express.json());

app.post('/api/balance', async (req, res) => {
  const { provider, apiKey, customUrl } = req.body;
  try {
    let balance = null;
    let currency = 'CNY';
    if (provider === 'openai') {
      const response = await fetch('https://api.openai.com/dashboard/billing/credit_grants', {
        headers: { 'Authorization': `Bearer ${apiKey}` }
      });
      balance = (await response.json()).total_available || 0;
      currency = 'USD';
    } else if (provider === 'deepseek') {
      const response = await fetch('https://api.deepseek.com/user/balance', {
        headers: { 'Authorization': `Bearer ${apiKey}` }
      });
      balance = (await response.json()).balance_info?.total_balance || 0;
    } else if (provider === 'moonshot') {
      const response = await fetch('https://api.moonshot.cn/v1/users/me/balance', {
        headers: { 'Authorization': `Bearer ${apiKey}` }
      });
      balance = (await response.json()).data?.balance || 0;
    }
    res.json({ success: true, balance, currency });
  } catch (err) {
    res.status(500).json({ success: false, error: err.message });
  }
});
app.listen(3000, () => console.log('Proxy running on port 3000'));

方案三：Python Flask

from flask import Flask, request, jsonify
from flask_cors import CORS
import requests
app = Flask(__name__)
CORS(app)

@app.route('/api/balance', methods=['POST'])
def get_balance():
    data = request.json
    provider = data.get('provider')
    api_key = data.get('apiKey')
    try:
        balance = None
        currency = 'CNY'
        if provider == 'openai':
            resp = requests.get('https://api.openai.com/dashboard/billing/credit_grants',
                headers={'Authorization': f'Bearer {api_key}'})
            balance = resp.json().get('total_available', 0)
            currency = 'USD'
        elif provider == 'deepseek':
            resp = requests.get('https://api.deepseek.com/user/balance',
                headers={'Authorization': f'Bearer {api_key}'})
            balance = resp.json().get('balance_info', {}).get('total_balance', 0)
        elif provider == 'moonshot':
            resp = requests.get('https://api.moonshot.cn/v1/users/me/balance',
                headers={'Authorization': f'Bearer {api_key}'})
            balance = resp.json().get('data', {}).get('balance', 0)
        return jsonify({'success': True, 'balance': balance, 'currency': currency})
    except Exception as e:
        return jsonify({'success': False, 'error': str(e)}), 500

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=3000)

部署步骤

  1. 选择上述任一方案部署代理服务
  2. 确保代理地址可通过 HTTPS 访问
  3. 在"同步设置"中填入代理地址
  4. 为每个模型填写对应的 API Key
  5. 点击"测试连接"验证配置
